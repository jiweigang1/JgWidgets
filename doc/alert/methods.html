$.jgAlert(message, fn, cssClass):
  message:alert内容
  fn:回调函数，点击确定后执行
  cssClass:指定弹出样式，一般用于指定Alert内容颜色
  弹出Alert框

$.jgConfirm(message, fn, cssClass):
  message:confirm内容
  fn:回调函数，点击确定后执行
  cssClass:指定弹出样式，一般用于指定confirm内容颜色
  弹出Confirm框

$.jgLog(message, type, wait):
  message:提示内容
  type:[success,error]
  wait:提示停留时间(默认10s)
  弹出提示框

$.jgSuccess(message, wait):
  message:提示内容
  wait:提示停留时间(默认10s)
  弹出成功提示框

$.jgError(message, wait):
  message:提示内容
  wait:提示停留时间(默认10s)
  弹出错误提示框
